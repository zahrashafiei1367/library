CREATE TABLE member(id int not null PRIMARY key AUTO_INCREMENT,name varchar(30),family varchar(30),username varchar(15),password varchar(40),isAdmin boolean not null);
CREATE TABLE book(id int not null PRIMARY key AUTO_INCREMENT,name varchar(30),writer varchar(30),publisher varchar(30) not null,category varchar(15),isAvailable boolean not null);
CREATE TABLE borrowing(id int not null PRIMARY key AUTO_INCREMENT,dateOfBorrowing date not null,dateOfReturning date not null,bookId int not null,userId int not null,CONSTRAINT FK_borrowingMember FOREIGN KEY (userId)  REFERENCES member(id),CONSTRAINT FK_borrowingBook FOREIGN KEY (bookId)  REFERENCES book(id));
